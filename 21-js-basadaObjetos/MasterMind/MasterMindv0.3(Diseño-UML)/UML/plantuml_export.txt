@startuml
title Mastermind
object game {
  COLORS,
  COMBINATION_LENGTH,
  MAX_ATTEMPS
  play(),
  printResults(),
  isLooser()
}

object yesOrNoContinue{
answer,
askContinue()
isAfirmative(),
isNegative()
}

object combination{
proposedCombinations,
readCombination()
isCorrectLenght(),
isCorrectColor(),
isRepeatColor()
}

object result{
blacks,
whites
}

object secret{
calculateResult()
isWinner()
}

game .left.> yesOrNoContinue
game *-down-> result
game *-down-> secret 
game *-down-> combination 
secret -right-> result
secret -left-> combination

@enduml
